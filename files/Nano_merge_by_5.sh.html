#!/usr/bin/env bash

#

# Merges "USIMC1951-page1.pdf" through "USIMC1951-page177.pdf"

# into batches of 5 pages each.

#

# Output files are named sequentially:

# USIMC1951_1.pdf, USIMC1951_2.pdf, ... etc.

#

# I added a section to remove the original single-page PDFs after merging

total_pages=177     # Change if you have fewer or more input PDFs.

chunk=5             # Number of pages to combine per merged PDF.

start=1             # Starting page for each chunk.

batch_index=1       # Used to name each merged PDF sequentially.

while [ $start -le $total_pages ]; do

  end=$(( start + chunk - 1 ))

  if [ $end -gt $total_pages ]; then

    end=$total_pages

  fi

  # Build a list of PDFs to merge for this batch

  page_files=""

  for (( i=$start; i<=$end; i++ )); do

    page_files="$page_files USIMC1951-page${i}.pdf"

  done

  # Construct output filenames in the format "USIMC1951_X.pdf".

  output_pdf="USIMC1951_${batch_index}.pdf"

  echo "Merging pages $start through $end into $output_pdf ..."

  # Merge the PDFs

  pdfunite $page_files "$output_pdf"

  # Prepare for next batch

  start=$(( end + 1 ))

  batch_index=$(( batch_index + 1 ))

done

echo "Done merging in batches of 5!"

# -------------------------------------------------------------

# Delete the original single-page PDFs afterward

rm USIMC1951-page*.pdf

# -------------------------------------------------------------
